create table POKEMON(
numPokemon int primary key AUTO_INCREMENT,
IDPokemon int,
nomPokemon char(50),
type1Pokemon int,
type2Pokemon int,
total int,
viePokemon int,
atqPokemon int,
defPokemon int,
speAtqPokemon int,
speDefPokemon int,
spdPokemon int,
generationPokemon int,
legendary boolean,
constraint FR_Type1 Foreign Key (type1Pokemon) references TYPE (numType),
constraint FR_Type2 Foreign Key (type2Pokemon) references TYPE (numType)
);

create Table TYPE(
numType int primary Key AUTO_INCREMENT,
labelType char(20));
